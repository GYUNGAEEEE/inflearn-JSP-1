# 26. JSP와 데이터 베이스 연동(2)
## (2) DAO 패턴을 적용하여 데이터 베이스 연동하기
→ database 디렉터리 참고!
```
CREATE TABLE MEMBER (
    ID VARCHAR2(20) PRIMARY KEY,
    PASS1 VARCHAR2(20),
    EMAIL VARCHAR2(50),
    TEL VARCHAR2(20),
    HOBBY VARCHAR2(60),
    JOB VARCHAR2(15),
    AGE VARCHAR2(10),
    INFO VARCHAR2(500)
);
```
